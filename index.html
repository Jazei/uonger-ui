    # uonger-ui 组件库

    # 快速开始

    # 1.安装组件库

    npm i uonger-ui

    # 2.引用组件库

    // 全部引入
import 'uonger-ui/dist/css/index.css'
import UUI from 'uonger-ui'
Vue.use(UUI)

//按需引用
import 'uonger-ui/dist/css/demo.css'
import {Demo} from 'uonger-ui'
Vue.use(Demo)
